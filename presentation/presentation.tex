% article の引用フォーマット
\newbibmacro{cite:article}{%
  \usebibmacro{journal}%
  \setunit{\addcomma\space}%
  \printfield{volume}%
  \setunit{\addcomma\space}%
  \printfield{pages}%
  \setunit{\space}%
  \printfield{year}%
}

% book の引用フォーマット
\newbibmacro{cite:book}{%
  \printfield{author}%
  \setunit{\addcomma\space}%
  \printfield{title}%
  \setunit{\addcomma\space}%
  \printfield{year}%
}
